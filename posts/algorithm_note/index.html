冒泡排序（Bubble Sort）

冒泡排序的本质在于==交换==，即每次通过交换的方式把当前剩余元素的最大值移动到一端，而当剩余元素减少为0时，排序结束。

#include <iostream>

using namespace std;

int main()
{
    int a[5] = {3, 1, 4, 5, 2};
    for (int i = 1; i <= 4; i++) // 进行n-1趟
    {
        // 从第i趟时从a[0]到a[n-i-1]都与它们下一个数比较
        for (int j = 0; j < 5 - i; j++)
        {
            if (a[j] > a[j + 1]) // 如果左边的数更大，则交换a[j]和a[j+1]
            {
                int Temp = a[j];
                a[j] = a[j + 1];
                a[j + 1] = Temp;
            }
        }
    }
    for (int i = 0; i < 5; i++)
    {
        cout << a[i];
    }

    return 0;
}

重点在于函数的==递归==调用

#include <iostream>

using namespace std;

int F(int n)
{
    if (n == 0)
        return 1;
    else
        return F(n - 1) * n;
}

int main()
{
    int n;
    cin >> n;
    cout << F(n) << endl;
    return 0;
}

害死人不偿命的（3n+1）猜想

对任何一个自然数n，如果它是偶数，那么把它砍掉一半；如果它是奇数，那么把（3n+1）砍掉一半，这样一直反复砍下去，最后一定在某一步得到n=1

# include<cstdio>

int main(){
	int n,step = 0;
	scanf("%d",&n);
	while(n != 1){
		if(n%2 == 0)
			n = n / 2;
		else
			n = (3*n+1) /2;
		step+=1;
	}
	printf("%d\n",step);
	return 0;
}

挖掘技术哪家强

在第1行给出不超过e5的正整数n，即参赛人数，随后n行，每行给出以为参赛者的信息和成绩，包括其所代表的学校的编号及其比赛成绩（百分制），在最后一行给出总得分最高的学校的编号及其总分。

# include<cstdio>
const int maxn = 100010; // 参赛人数n不超过e5的正整数 
int school[maxn] = {0};

int main(){
	int n,schID,score;
	scanf("%d\n",&n); // 记录参赛人数
	for(int i =0; i<n;i++){
		scanf("%d %d",&schID,&score);  //学校ID，分数 
		 school[schID]+=score; // 学校schID的总分增加score 
	} 
	int k=1,Max=-1; // 最高得分的学校ID以及其总分 
	for(int i =1;i<=n;i++){ // 从所有学校中选出总分最高的一个 
		if(school[i]>Max){
			Max = school[i];
			k = i;	
		}
	}
	printf("%d %d",k,Max); //输出最高总分的学校ID及其总分 
	return 0;
}

找x

输入一个数n（1<=n<=200），然后输入n个数值各不相同的数，再输入一个值x，输出这个值在这个数组中的下标（从0开始，若不在数组中则输出-1）

# include<cstdio>

const int maxn = 210;
int a[maxn];

int main(){
	int n,x;
	while(scanf("%d",&n)!=EOF){
		for(int i=0;i<n;i++){
			scanf("%d",&a[i]);
		}
		scanf("%d",&x);//输入欲查询的数 
		int k; //下标
		for(k=0;k<n;k++){ //遍历数组 
			if(a[k]==x){
				printf("%d\n",k);
				break;
			}
		}
		if(k==n){
			printf("-1\n");
		}
	}
	return 0; 
}
